코드 예제

기본 사용법

// API 인스턴스 생성
const api = new AllTourismAPI({
    serviceKey: 'YOUR_API_KEY'
});

// 관광지 검색
const result = await api.areaBasedList({
    contentTypeId: '12',
    areaCode: '1',
    numOfRows: '10'
});

console.log(result);

위치 기반 검색

// 현재 위치 기준 주변 관광지
const nearbyPlaces = await api.areaBasedList({
    contentTypeId: '12',
    userLat: '37.5665',
    userLng: '126.9780',
    radius: '5'
});

// 거리 정보가 자동으로 추가됨
nearbyPlaces.data.items.forEach(item => {
    console.log(`${item.title}: ${item.meta.distanceText}`);
});

배치 요청

// 여러 요청을 동시에 처리
const batchResult = await api.batchRequest([
    { operation: 'areaBasedList', params: { contentTypeId: '12', areaCode: '1' } },
    { operation: 'searchKeyword', params: { keyword: '경복궁' } },
    { operation: 'areaCode', params: {} }
]);

console.log(batchResult);
